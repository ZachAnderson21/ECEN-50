syms V2 V3 V5

eq1 = (V2 - Vs1)/R1 + (V2 - V3)/R3 + (V2 - V5)/R5 + V2/R9 == 0;

V3_val = double(solutions.V3);
V5_val = double(solutions.V5);

% Thevenin equivalent voltage Vth

% Display
fprintf('The Thevenin Voltage (Vth) is %.4f V\n', Vth);
